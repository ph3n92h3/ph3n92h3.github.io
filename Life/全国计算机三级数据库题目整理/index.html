ON
(    NAME = 'Studentdb_Data',
     FILENAME = 'E:\Studentdb.mdf',
     SIZE = 3MB,
     MAXSIZE = 50MB,
     FILEGROWTH = 10%   /*超过初始大小后文件增长率*/
)
LOG ON
(    NAME = 'Studentdb_Log',
     FILENAME = 'E:\Studentdb.ldf',
     SIZE = 2MB,
     MAXSIZE = 5MB,
     FILEGROWTH = 1MB   /*超过初始大小后文件增长率*/

    添加和删除数据库文件

    ALTER DATABASE database
{
    ADD FILE filename [TO FILEGROUP filegroup_name]
    ADD LOG FILE filename
    ADD FILEGROUP filegroup_name

    REMOVE FILE logical_file_name
    REMOVE FILEGROUP filegroup_name

    MODIFY FILE filespec
    MODIFY NAME=new_dbname

    SET <optionspec> [,...n] [WITH <termination>]

    COLLATE <collation_name>
}

    扩大指定文件的大小

    ALTER DATABASE STU_DB MODIFY (NAME=student_data, SIZE=8MB)

    添加新的数据文件

    ALTER DATABASE STU_DB ADD FILE (NAME=student_data2, FILENAME=‘E:\Data\student_data2.ndf’, SIZE=6MB, FILEGROWTH=0)

    （用户自定义）存储过程 PROCEDURE

    删除：

    DROP PROCEDURE p_custbuy
DROP PROCEDURE p_custcout
    例一

    创建：

    CREATE PROCEDURE p_custbuy @area varchar(20)='长沙岳麓区', @Price money
AS
SELECT ... FROM ... JOIN ...
    WHERE Address=@area AND SaleUnitPrice>@Price

    执行：

    EXEC p_custbuy @Price=1000
    例二

    创建：

    CREATE PROCEDURE p_custcout @area varchar(20), @sex char, @count int output, @avg_age int output
AS
    SELECT @count=COUNT(*), @ave_age=AVG(YEAR(GETDATE())-YEAR(BIRTHDATE))
        FROM Table_Customer
        WHERE Address=@area AND Sex=@sex

    执行：

    DECLARE @x int, @y int
EXEC p_custcount '长沙岳麓区', 'F', @x output, @y output
SELECT @x AS 人数, @y AS 平均年龄

    用户自定义函数 FUNCTION

    删除：

    DROP FUNCTION f_GoodsCount
DROP FUNCTION f_GoodsInfo
DROP FUNCTION f_GoodsDatails
    标量函数

    创建：

    CREATE FUCTION dbo.f_GoodsCount(@class varchar(10))
   RETURN int
AS
BEGIN
   DECLARE @x int
   SELECT @x=count(*)
        FROM Table_GoodsClass a JOIN Table_Goods b
        ON a.GoodsClassID=b.GoodsClassID
        WHERE GoodsClassName=@class
   RETURN @x
END

    调用：

    SELECT GoodsName AS 商品名, dbo.f_GoodsCount('服装') AS 种类数 FROM ... WHERE ...
    内嵌表值函数

    创建：

    CREATE FUCTION f_GoodsInfo(@class char(10))
   RETURNS TABLE
AS
    RETURN(
        SELECT GoodName,SaleUnitPrice
            FROM Table_GoodClass a JOIN Table_Goods b ON a.GoodsClassID=b.GoodsClassID
            WHERE GoodClassName=@class)

    调用：

    SELECT * FROM dbo.f_GoodsInfo('服装')
    多语句表值函数

    创建：

    CREATE FUNCTION f_GoodsDetails(@class varchar(20))
RETURNS @f_GoodsDetails TABLE(
    GoodName varchar(50),
    SaleUnitPrice money,
    ProductionDate datetime,
    种类数 int
)
AS
BEGIN
    INSERT INTO @f_GoodsDetails
    SELECT GoodName, SaleUnitPrice, ProductionDate, dbo.f_GoodsDetails(@class)
        FROM Table_GoodClass a JOIN Table_Goods b ON a.GoodsClassID=b.GoodsClassID
        WHERE GoodsClassName=@class
        RETURN
END

    调用：

    SELECT * FROM dbo.f_GoodsInfo('服装')

    触发器 TRIGGER

    创建
    CREATE TRIGGER trigger_name ON
    { table | view }
    [WITH ENCRYPTION]
    { FOR | AFTER | INSTEAD OF }    /*FOR或AFTER：后触发；INSTEAD OF：前触发*/
    { [ INSERT ] [ , ] [ UPDATE ] [ , ] [ DELETE ] }
AS
    sql_statement[... n]

    后触发器：维护数据完整性

    CREATE TRIGGER UnitPriceConsistent ON Table_PriceHistory FOR INSERT, UPDATE
AS
    DECLARE @NewPrice money
    SELECT @NewPrice=SaleUnitPrice FROM inserted
    UPDATE Table_Goods SET SaleUnitPrice=@NewPrice
        WHERE GoodsID IN (SELECT GoodsID FROM inserted)

    前触发器：阻止非法操作

    CREATE TRIGGER DeleteCust ON Table_Customer INSTEAD OF DELETE
AS
   IF NOT EXISTS(SELECT * FROM deleted WHERE CardID IN (SELECT CardID FROM Table_Card WHERE Score>=500))
   DELETE FROM Table_Customer WHERE CardID IN (SELECT CardID FROM deleted)
    删除
    DROP TRIGGER OperateCon
DROP TRIGGER UnitPriceConsistent
DROP TRIGGER DeleteCust

    游标 cursor

    声明
    DECLARE cursor_name CURSOR FOR (select_statement)
    打开
    OPEN cursor_name
    提取
    FETCH FROM cursor_name [INTO @variable_name[, ...n]]    /*@@FETCH_STATUS=0时表示FETCH语句执行成功*/
    关闭
    CLOSE cursor_name
    释放
    DEALLOCATE cursor_name

    安全管理

    登录账户

    CREATE LOGIN SQL_User1 WITH PASSWORD='a1b2c3XY'     /*创建*/
ALTER LOGIN SQL_User1 WITH PASSWORD='a4b5c6XY'      /*修改*/
ALTER LOGIN SQL_User3 WITH NAME=NewUser             /*删除*/
/*有了登录账户只能连接到SQL服务器，不具有访问数据库的权限*/

    数据库用户

    CREATE USER user_name [FOR|FROM] LOGIN login_name
DROP USER user_name

/*Guest用户*/
GRANT CONNECT TO guest
REVOKE CONNECT TO guest

    权限

    六种：SELECT, INSERT, UPDATE, DELETE, REFERENCES, EXECUTE

    授权
    GRANT 对象权限 ON 对象 TO （主体：数据库用户名或角色） [WITH GRANT OPTION]  /*是否允许将该权限授予其他主体*/
    拒绝权限
    DENY 对象权限 ON 对象 TO （主体：数据库用户名或角色） [CASCADE] [AS 主体]   /*CASCADE: 同时拒绝这个主体曾经授予其他主体的权限*/
    收权
    REVOKE 对象权限 ON 对象 TO （主体：数据库用户名或角色） [CASCADE] [AS 主体] /*CASCADE: 同时收回这个主体曾经授予其他主体的权限*/
