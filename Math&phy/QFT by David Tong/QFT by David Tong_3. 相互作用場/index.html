    зӣёдә’дҪңз”Ёе ҙ

    жң¬з« жҖқи·Ҝ

    1. зӣёдә’дҪңз”Ёз№ӘжҷҜ
    2. жұӮи§Јзӣёдә’дҪңз”Ёз№ӘжҷҜдёӢзҡ„йҒӢеӢ•ж–№зЁӢ \(\rightarrow\) Dyson е…¬ејҸ/жј”еҢ–з®—з¬Ұ
    \[\left|\psi(t)\right\rangle_I=U(t,t_0)\left|\psi(t,t_0)\right\rangle_I,\ U(t,t_0)=T\exp\left(-\mathrm{i}\int^t_{t_0}H_I(t')\mathrm{d}t'\right)\]
    1. еёҢжңӣеҫ—еҲ°зҡ„еҪўејҸпјҡ\(\left\langle f\right|a_1^{\dagger}a_2^{\dagger}\dots a_1a_2\left|i\right\rangle\) еҚі normal ordering \(\rightarrow\) Wick е®ҡзҗҶ
    2. жҺЁеҜјеҮә Feynman и§„еҲҷ

    дҪҝз”Ёеҫ®ж“ҫзҗҶи«–

    з”ұйҮҸз¶ұеҲҶжһҗзҹҘпјҢе°Қж–ј \(\mathcal{L}=\frac{1}{2}\partial_{\mu}\phi\partial^{\mu}\phi-\frac{1}{2}m^2\phi^2-\sum_{n\leq 3}\frac{\lambda_n}{n!}\phi^n\)пјҢеңЁдҪҺиғҪжғ…жіҒдёӢпјҢеҸӘжңү \(n=3\text{ or }4\) жҳҜйңҖиҰҒиҖғж…®зҡ„гҖӮ

    зӣёдә’дҪңз”Ёз№ӘжҷҜ

    \[ H=H_0+H_{\mathrm{int}} \]
    \[ H_{\mathrm{int}}\equiv(H_{\mathrm{int}})_I=\mathrm{e}^{\mathrm{i}H_0t}(H_{\mathrm{int}})_S\mathrm{e}^{-\mathrm{i}H_0t} \]
    \[ \mathrm{i}\frac{\mathrm{d}\left|\psi\right\rangle_I}{\mathrm{d}t}=H_I(t)\left|\psi\right\rangle_I \]

    з¬¬дёҖж¬ЎжұӮи§ЈжҢҜе№…

    иӨҮжЁҷйҮҸе ҙе’ҢеҜҰжЁҷйҮҸе ҙиҖҰеҗҲпјҡ\(H_{\mathrm{int}}=g\int\mathrm{d}^3x\ \psi^{\dagger}\psi\phi\)

    \[ \left|i\right\rangle=\sqrt{2E_{\vec{p}}}\ a_{\vec{p}}^{\dagger}\left|0\right\rangle,\ \left|f\right\rangle=\sqrt{4E_{\vec{q}_1}E_{\vec{q}_2}}\ b_{\vec{q}_1}^{\dagger}c_{\vec{q}_2}^{\dagger}\left|0\right\rangle \]

    иЁҲз®—еҲ°дёҖйҡҺй …пјҡ

    \[ \begin{aligned} \left\langle f\right|U(+\infty,-\infty)\left|i\right\rangle&\equiv\left\langle f\right|1-\mathrm{i}\int\mathrm{d}t\ H_I(t)\left|i\right\rangle=\left\langle f\right|-\mathrm{i}\int\mathrm{d}t\ H_I(t)\left|i\right\rangle\\ &=-\mathrm{i}\left\langle f\right|\int\mathrm{d}t\ H_I(t)\left|i\right\rangle\\ &=-\mathrm{i}\left\langle f\right|\int\mathrm{d}t\ d\int\mathrm{d}^3x\ \psi^{\dagger}(x)\psi(x)\phi(x)\left|i\right\rangle\\ &=-\mathrm{i}g\left\langle f\right|\int\mathrm{d}^4x\ \psi^{\dagger}(x)\psi(x)\phi(x)\left|i\right\rangle\\ &=-\mathrm{i}g\left\langle f\right|\int\mathrm{d}^4x\ \psi^{\dagger}(x)\psi(x)\int\frac{\mathrm{d}^3k}{(2\pi)^3}\frac{\sqrt{2E_{\vec{p}}}}{\sqrt{2E_{\vec{k}}}}a_{\vec{k}}a_{\vec{p}}^{\dagger}\mathrm{e}^{-\mathrm{i}k\cdot x}\left|0\right\rangle\\ \end{aligned} \]

    зҲІд»ҖйәҪеҸӘеҜ«еҮәдҫҶеҚҠеҖӢ \(\phi(x)\) зҡ„еұ•й–ӢејҸпјҹеӣ зҲІеҸҰеӨ–еҚҠеҖӢпјҡ

    \[ \begin{aligned} & \left\langle f\right|\int\mathrm{d}^4x\ \psi^{\dagger}(x)\psi(x)\int\frac{\mathrm{d}^3k}{(2\pi)^3}\frac{\sqrt{2E_{\vec{p}}}}{\sqrt{2E_{\vec{k}}}}a_{\vec{k}}^{\dagger}a_{\vec{p}}^{\dagger}\mathrm{e}^{\mathrm{i}k\cdot x}\left|0\right\rangle\\ = & \left\langle 0\right|a_{\vec{k}}^{\dagger}\sqrt{4E_{\vec{q}_1}E_{\vec{q}_2}}b_{\vec{q}_1}c_{\vec{q}_2}\int\mathrm{d}^4x\ \psi^{\dagger}(x)\psi(x)\int\frac{\mathrm{d}^3k}{(2\pi)^3}\frac{\sqrt{2E_{\vec{p}}}}{\sqrt{2E_{\vec{k}}}}a_{\vec{p}}^{\dagger}\mathrm{e}^{\mathrm{i}k\cdot x}\left|0\right\rangle\text{пјҢзӣҙжҺҘз§»йҒҺеҺ»дәҶпјҢеӣ зҲІ aгҖҒbгҖҒc д№Ӣй–“зҡ„е°Қжҳ“еӯҗзӮә 0}\\ = & 0 \end{aligned} \]

    еӣһдҫҶз№јзәҢпјҡ

    \[ \begin{aligned} & -\mathrm{i}g\left\langle f\right|\int\mathrm{d}^4x\ \psi^{\dagger}(x)\psi(x)\int\frac{\mathrm{d}^3k}{(2\pi)^3}\frac{\sqrt{2E_{\vec{p}}}}{\sqrt{2E_{\vec{k}}}}a_{\vec{k}}a_{\vec{p}}^{\dagger}\mathrm{e}^{-\mathrm{i}k\cdot x}\left|0\right\rangle\\ \xlongequal{a_{\vec{k}}a_{\vec{p}}^{\dagger}=a_{\vec{p}}^{\dagger}a_{\vec{k}}+(2\pi)^3\delta^{(3)}(\vec{p}-\vec{k})} & -\mathrm{i}g\left\langle f\right|\int\mathrm{d}^4x\ \psi^{\dagger}(x)\psi(x)\int\frac{\mathrm{d}^3k}{(2\pi)^3}\frac{\sqrt{2E_{\vec{p}}}}{\sqrt{2E_{\vec{k}}}}(2\pi)^3\delta^{(3)}(\vec{p}-\vec{k})\mathrm{e}^{-\mathrm{i}k\cdot x}\left|0\right\rangle\\ = & -\mathrm{i}g\left\langle f\right|\int\mathrm{d}^4x\ \psi^{\dagger}(x)\psi(x)\mathrm{e}^{-\mathrm{i}p\cdot x}\left|0\right\rangle\\ = & -\mathrm{i}g\left\langle 0\right|\int\frac{\mathrm{d}^4x\mathrm{d}^3k_{1}\mathrm{d}^3k_{2}}{(2\pi)^6}\frac{\sqrt{E_{\vec{q}_1}E_{\vec{q}_2}}}{\sqrt{E_{\vec{k}_1}E_{\vec{k}_2}}}b_{\vec{q}_1}c_{\vec{q}_2}c_{\vec{k}_1}^{\dagger}b_{\vec{k}_2}^{\dagger}\mathrm{e}^{\mathrm{i}(k_1+k_2-p)\cdot x}\left|0\right\rangle\\ = & -\mathrm{i}g\left\langle 0\right|\int\mathrm{d}^4x\ \mathrm{e}^{\mathrm{i}(q_1+q_2-p)\cdot x}\left|0\right\rangle\\ = & -\mathrm{i}g(2\pi)^4\delta^{(4)}(q_1+q_2-p) \end{aligned} \]

    Wick е®ҡзҗҶиЁҲз®—ж ёеӯҗж•Је°„ \(\psi\psi\to\psi\psi\)

    \[ \left|i\right\rangle=\sqrt{2E_{\vec{p}_1}}\sqrt{2E_{\vec{p}_2}}b_{\vec{p}_1}^{\dagger}b_{\vec{p}_2}^{\dagger}\left|0\right\rangle\equiv\left|p_1,p_2\right\rangle,\ \left|f\right\rangle=\sqrt{2E_{\vec{p}_1'}}\sqrt{2E_{\vec{p}_2'}}b_{\vec{p}_1'}^{\dagger}b_{\vec{p}_2'}^{\dagger}\left|0\right\rangle\equiv\left|p_1',p_2'\right\rangle \]

    жҢҜе№…пјҡ

    \[ \frac{(-\mathrm{i}g)^2}{2}\int\mathrm{d}^4x_1\mathrm{d}^4x_2T\left(\psi^{\dagger}(x_1)\psi(x_1)\phi(x_1)\psi^{\dagger}(x_2)\psi(x_2)\phi(x_2)\right) \]

    з”ұ Wick е®ҡзҗҶеҒҡжӣҝжҸӣпјҢд№ӢеҫҢжңғз•ҷдёӢзҡ„йқһйӣ¶й …еҸӘжңү \(:\psi^{\dagger}(x_1)\psi(x_1)\psi^{\dagger}(x_2)\psi(x_2):\overbrace{\phi(x_1)\phi(x_2)}\)пјҢеӣ зҲІе…ҲиҰҒж№®ж»…еҲқж…Ӣзҡ„е…©еҖӢж ёеӯҗпјҢеҶҚз”ҹжҲҗжЁЎж…Ӣзҡ„е…©еҖӢж ёеӯҗпјҢе°Қж–је…¶д»–зҡ„й …пјҢжңғз”ўз”ҹд»Ӣеӯҗ which дёҚжҳҜжҲ‘еҖ‘жғіиҰҒзҡ„гҖӮ

    жіЁ рҹҗ– пјҡйӮЈеҖӢејҸеӯҗ рҹҰҒ дёҠж–№зҡ„еӨ§жӢ¬иҷҹжҳҜ \overbrace{}

    е…¶д»–зҡ„е°ұжӯЈеёёйҒӢз®—жҮүи©ІжІ’д»ҖйәҪе•ҸйЎҢеҗ§пјҢжңүйҖҷйәҪдёҖй … \(\left\langle0\right|\psi(x)\left|p\right\rangle=\mathrm{e}^{-\mathrm{i}p\cdot x}\)пјҢе®ғдҫҶиҮӘпјҡ

    \[ \begin{aligned} \left\langle0\right|\psi(x)\left|p\right\rangle & = \left\langle0\right|\int\frac{\mathrm{d}^3p}{(2\pi)^3}\frac{1}{\sqrt{2E_{\vec{p}}}}\left(b_{\vec{p}}\mathrm{e}^{-\mathrm{i}p\cdot x}+c_{\vec{p}}^{\dagger}\mathrm{e}^{+\mathrm{i}p\cdot x}\right)\sqrt{2E_{\vec{p}}}b_{\vec{p}}^{\dagger}\left|0\right\rangle \end{aligned} \]

    еҫҢйқўдёҚз”ЁжҲ‘з№јзәҢеҜ«дәҶзҪ·гҖӮйӮ„жғіжҢҮеҮәдёҖй»һпјҢжҠҪиұЎеҮәдҫҶе°ұжҳҜйҖҷйәҪеҖӢжқұиҘҝпјҡ

    \[ \begin{aligned} & \int_{-\infty}^{\infty}\mathrm{d}x\ f(x)\delta(x-x_1)\delta(x-x_2)\\ \xlongequal{F(x)\equiv f(x)\delta(x-x_1)} & \int_{-\infty}^{\infty}\mathrm{d}x\ F(x)\delta(x-x_2)\\ = & F(x_2)\\ = & f(x_2)\delta(x_2-x_1)=f(x_1)\delta(x_2-x_1) \end{aligned} \]

    Feynman ең–

    еғҸиҝ‘е№ҙдҫҶзҡ„зЎ…иҠҜзүҮдёҖжЁЈпјҢFeynman ең–зӮәеӨ§иЎҶеё¶дҫҶдәҶиЁҲз®—гҖӮвҖ”вҖ” Julian Schwinger

    \[H_{\mathrm{int}}=g\int\mathrm{d}^3x\ \psi^{\dagger}\psi\phi\]

    жҖҺйәҪз•«

    вҖңжҖҺйәҪз•«вҖқжҢҮжҖҺйәҪе°ҮжғіиҰҒиЁҲз®—зҡ„зү©зҗҶйҒҺзЁӢиҪүеҢ–зҲІ Feynman ең–пјҢд»ҘйҒӢз”ЁзӣёжҮүзҡ„иҰҸеүҮеҜ«еҮәж•Је°„жҢҜе№…иЎЁйҒ”ејҸгҖӮ

    1. еҲқж…ӢгҖҒжң«ж…Ӣзҡ„жҜҸеҖӢзІ’еӯҗе°ҚжҮүдёҖжўқе…Ҙе°„гҖҒеҮәе°„з¶«пјҲеӨ–йғЁз¶«пјүпјӣ
    2. д»ӢеӯҗпјҲfrom еҜҰжЁҷйҮҸе ҙпјүз”Ёй»һз¶«пјҢж ёеӯҗпјҲfrom иӨҮжЁҷйҮҸе ҙпјүз”ЁеҜҰз¶«пјӣпјҲйҖҷиЈҸзҡ„д»Ӣеӯҗе’Ңж ёеӯҗдёҚжҳҜзҸҫеҜҰз”ҹжҙ»дёӯзҡ„йӮЈеҖӢпјү
    3. зӮәжҜҸжўқз¶«еҒҮиЁӯдёҖеҖӢеӢ•йҮҸпјҲеӨ§е°Ҹе’Ңж–№еҗ‘пјүпјҢйЎһдјјж–јйӣ»и·Ҝдёӯзҡ„вҖңеҸғиҖғж–№еҗ‘вҖқпјӣ
    4. еҜҰз¶«пјҲж ёеӯҗпјүдёҠжЁҷжіЁз®ӯй ӯд»ҘжҢҮжҳҺе…¶иҚ·пјҡ
    5. еҲқж…Ӣ \(\psi\) з”Ёе…Ҙе°„з®ӯй ӯпјҢеҲқж…Ӣ \(\bar\psi\) з”ЁеҮәе°„з®ӯй ӯпјӣ
    6. жң«ж…Ӣ \(\psi\) з”ЁеҮәе°„з®ӯй ӯпјҢжң«ж…Ӣ \(\bar\psi\) з”Ёе…Ҙе°„з®ӯй ӯпјӣ
    7. зёҪжҳҜпјҢ\(\psi\) жҳҜжҜ”ијғиҮӘз„¶зҡ„пјҢ\(\bar\psi\) жҳҜеҸҚи‘—зҡ„пјӣ
    8. е°Қж–јзҸҫеңЁзҡ„жғ…жіҒпјҢз”Ёдёүеғ№зҡ„й Ӯй»һе°Үд»–еҖ‘йҖЈжҺҘиө·дҫҶгҖӮ

    еңЁз•«ең–йҒҺзЁӢдёӯи«ӢиҖғж…®е°ҚзЁұжҖ§гҖӮ

    жҖҺйәҪз®—пјҡFeynman жі•еүҮ

    вҖңжҖҺйәҪз®—вҖқжҢҮеңЁе·Із¶“еӯҳеңЁдёҖеҖӢ Feynman ең–д№ӢеҫҢжҖҺйәҪе°Үе®ғиҪүеҢ–зҲІе°ҚжҮүзү©зҗҶйҒҺзЁӢзҡ„ж•Је°„жҢҜе№…гҖӮ

    1. жҜҸжўқеҶ…йғЁз¶«ж·»еҠ дёҖеҖӢеӢ•йҮҸ \(\vec{k}\)пјӣпјҲеҶ…йғЁз¶«жҢҮйқһе…Ҙе°„гҖҒеҮәе°„зҡ„з¶«пјү
    2. еңЁжҜҸеҖӢй Ӯй»һдёҠпјҢеҸҜд»ҘеҜ«дёӢеӣ еӯҗ \((-\mathrm{i}g)(2\pi)^4\delta^{(4)}\left(\sum_ik_i\right)\)пјҢе…¶еҜҰе°ұжҳҜеӢ•йҮҸе®ҲжҒҶпјӣ
    3. е°Қж–јжҜҸеҖӢеҶ…йғЁзҡ„й»һз¶«пјҢжҲ‘еҖ‘зҹҘйҒ“йҖҷжҳҜдёҖеҖӢеӢ•йҮҸзӮә \(\vec{k}\) зҡ„д»ӢеӯҗпјҢеҸҜд»ҘеҜ«дёӢеӣ еӯҗ \(\int\frac{\mathrm{d}^2k}{(2\pi)^4}\frac{\mathrm{i}}{k^2-m^2+\mathrm{i}\epsilon}\)пјӣе°Қж–јж ёеӯҗпјҲеҜҰз¶«пјүпјҢд»Қз„¶жңүйҖҷеҖӢеӣ еӯҗпјҢдҪҶиЁҳеҫ—з”Ёж ёеӯҗзҡ„иіӘйҮҸгҖӮ
    4. жңүеңҲзҡ„жғ…жіҒпјҢйҖІиЎҢз©ҚеҲҶ \(\int\mathrm{d}^2k/(2\pi)^4\)

    йҖҷжҳҜе°Қж–јжҲ‘еҖ‘зҸҫеңЁиҰҒиҷ•зҗҶзҡ„е“ҲеҜҶй “йҮҸзҡ„ Feynman жі•еүҮпјҢе°Қж–јд»ҘеҫҢжңғиҰӢеҲ°зҡ„е…¶е®ғзІ’еӯҗпјҢжңғжңүйЎҚеӨ–зҡ„иҰҸеүҮгҖӮ

    жіЁ рҹҗ– пјҡдёҚиҖғж…®йқһйҖЈйҖҡең–пјҢдёҚиҖғж…®еӨ–йғЁз¶«дёҠжңүеңҲзҡ„йқһжҲӘж–·ең–гҖӮ

    • д»Ӣзҙ№е®Ң Feynman ең–д№ӢеҫҢе°ұжҳҜдёҖдәӣд»Ӣзҙ№жҖ§зҡ„еҶ…е®№пјҢе®ўи§ҖдёҠдҫҶиӘ¬е…¶дёӯдёҖдәӣжқұиҘҝиӘ¬зҡ„жҳҜдёҚеӨ жё…жҘҡзҡ„гҖӮз„ЎеҸҜжҢҮж‘ҳпјҢеӣ зҲІд»–еҖ‘йӮ„жңүдёӢдёҖеҖӢеӯёжңҹзҡ„иӘІзЁӢ вҖңAQFTпјҲй«ҳзӯүйҮҸеӯҗе ҙи«–пјүвҖқпјҢиҖҢжҲ‘и®ҖйҖҷжң¬и¬ӣзҫ©д№ҹжҳҜеҸӘжҳҜеңЁеӯёзҝ’ GR д№Ӣй–“еҒҡдёҖеҖӢиӘҝеҠ‘гҖӮжӣҙи©ізҙ°зҡ„еҶ…е®№зҗҶжҮүдёҚеңЁйҖҷиЈҸеҮәзҸҫгҖӮ

    дёҖеҖӢз©ҚеҲҶ

    дҪҹеӨ§зҲІеңЁз”ЁдёҖдәӣеҘҮеҰҷзҡ„ж–№жі•пјҢжҲ‘дёҚз®ЎдәҶпјҢжҲ‘зӣҙжҺҘжўҒжҳҶж·јдәҶ:

    \[ \begin{aligned} &\int^{\infty}_0\mathrm{d}k\ \frac{k}{k^2+m^2}\sin(kr)\\ = & \pi\times\left\{\frac{k}{k^2+m^2}\ \mathrm{e}^{\mathrm{i}kr}\text{ еңЁдёҠеҚҠе№ійқўжүҖжңүеҘҮй»һзҡ„з•ҷж•ёд№Ӣе’Ң}\right\}\\ = & \pi\times\mathrm{Res}\left.\left[\frac{k}{(k+\mathrm{i}m)(k-\mathrm{i}m)}\ \mathrm{e}^{\mathrm{i}kr}\right]\right|_{k=\mathrm{i}m}\\ = & \pi\times\lim_{k\to\mathrm{i}m}\left[(k-\mathrm{i}m)\cdot\frac{k}{(k+\mathrm{i}m)(k-\mathrm{i}m)}\ \mathrm{e}^{\mathrm{i}kr}\right]\\ = & \pi\times\frac{1}{2}\mathrm{e}^{-mr} \end{aligned} \]
    • дёӢдёҖз« д»Ӣзҙ№ Dirac ж–№зЁӢпјҢдј°иЁҲд»Јж•ёеҶ…е®№жңғжҜ”ијғеӨҡпјҢжҲ‘еҫҲжңҹеҫ…гҖӮрҹҳҖ
